#include <stdio.h>
#include <string.h>
int main()
{
	float a1;
	float b1;
	float a2;
	float b2;
	float p1;
	float p2;
	scanf( "%f%f%f%f", &a1, &b1, &a2, &b2 );
	p1 = a1 * b1 / 2;
	p2 = a2 * b2 / 2;
	if ( p1 > p2 ) {
		printf( "%s", "1" );
	} else {
		if ( p1 < p2 ) {
			printf( "%s", "2" );
		} else {
			printf( "%s", "0" );
		}
	}
	return 0;
}
